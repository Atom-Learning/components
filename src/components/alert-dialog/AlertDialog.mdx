```jsx
import { AlertProvider, Button, useAlert } from '@atom-learning/components'

const Component = () => {
  const { showAlert } = useAlert()

  const handleClick = () => {
    showAlert({
      title: 'Are you sure you want to delete this school?',
      description:
        'Removing this last selected school will remove all restrictions on the exam and make it available to all schools.',
      confirmActionText: 'Delete school',
      onAction: (result) => {
        if (result) console.log('delete school')
      }
    })
  }

  return <Button onClick={handleClick}>Click to show alert</Button>
}

const App = () => (
  <AlertProvider>
    <Component />
  </AlertProvider>
)

ReactDom.render(<App />, document.getElementById('root'))
```
